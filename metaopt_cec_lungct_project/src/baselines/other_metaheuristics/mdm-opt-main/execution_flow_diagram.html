Example: PLO (Polar Lights Optimizer) Evolution
def evolve(self, epoch):
    """PLO (Polar Lights Optimizer) evolution step"""
    # Calculate adaptive parameters
    progress_ratio = epoch / self.epoch
    x_mean = np.mean([agent.solution for agent in self.pop], axis=0)

    # Adaptive weights
    w1 = np.tanh((progress_ratio) ** 4)
    w2 = np.exp(-(2 * progress_ratio) ** 3)
    E = np.sqrt(progress_ratio)

    # Generate random permutation for collision pairs
    A = self.generator.permutation(self.pop_size)

    pop_new = []
    for idx in range(self.pop_size):
        # Aurora oval walk
        a = self.generator.uniform() / 2 + 1
        V = np.exp((1 - a) / 100 * epoch)
        LS = V

        # Levy flight movement component
        GS = self.levy(self.problem.n_dims) * (
            x_mean - self.pop[idx].solution +
            (self.problem.lb + self.generator.uniform(0, 1, self.problem.n_dims) *
             (self.problem.ub - self.problem.lb)) / 2
        )

        # Position update based on aurora oval walk
        pos_new = self.pop[idx].solution + (w1 * LS + w2 * GS) * \
                  self.generator.uniform(0, 1, self.problem.n_dims)

        # Particle collision mechanism
        for j in range(self.problem.n_dims):
            if (self.generator.random() < 0.05) and (self.generator.random() < E):
                pos_new[j] = self.pop[idx].solution[j] + \
                           np.sin(self.generator.random() * np.pi) * \
                           (self.pop[idx].solution[j] - self.pop[A[idx]].solution[j])

        # Ensure position is within bounds
        pos_new = self.correct_solution(pos_new)
        agent = self.generate_empty_agent(pos_new)
        pop_new.append(agent)
Example: RIME Evolution
def evolve(self, epoch):
    """RIME (RIME Ice Marine-predator Optimizer) evolution step"""
    # RIME-specific parameters
    rime_factor = (self.generator.random() - 0.5) * 2 * \
                  np.cos(np.pi * epoch / (self.epoch / 10)) * \
                  (1 - np.round(epoch * self.sr / self.epoch) / self.sr)

    ee = np.sqrt((epoch + 1) / self.epoch)

    # Normalize fitness values
    fits = np.array([agent.target.fitness for agent in self.pop]).reshape((1, -1))
    fits_norm = fits / np.linalg.norm(fits, axis=1, keepdims=True)

    LB = self.problem.lb
    UB = self.problem.ub
    pop_new = []

    for idx in range(self.pop_size):
        pos_new = self.pop[idx].solution.copy()

        for jdx in range(self.problem.n_dims):
            # Soft-rime search strategy
            if self.generator.random() < ee:
                pos_new[jdx] = self.g_best.solution[jdx] + \
                              rime_factor * (LB[jdx] + self.generator.random() *
                                           (UB[jdx] - LB[jdx]))

            # Hard-rime puncture mechanism
            if self.generator.random() < fits_norm[0, idx]:
                pos_new[jdx] = self.g_best.solution[jdx]

        pos_new = self.correct_solution(pos_new)
        agent = self.generate_empty_agent(pos_new)
        pop_new.append(agent)

📊 Progress Tracking
Each iteration tracks optimization progress, updates global best, and stores historical data. 
def track_optimize_step(self, pop, epoch, runtime):
    """Track optimization progress for current epoch"""
    # Update global best and worst
    current_best = self.get_best_agent(pop, self.problem.minmax)
    current_worst = self.get_worst_agent(pop, self.problem.minmax)

    # Store current epoch data in history
    self.history.list_current_best.append(current_best)
    self.history.list_current_worst.append(current_worst)
    self.history.list_epoch_time.append(runtime)
    self.history.list_global_best_fit.append(self.g_best.target.fitness)
    self.history.list_current_best_fit.append(current_best.target.fitness)

    # Calculate and store population diversity
    diversity = self.calculate_diversity(pop)
    self.history.list_diversity.append(diversity)

    # Optional: Store complete population (memory intensive)
    if self.problem.save_population:
        self.history.list_population.append([agent.copy() for agent in pop])

🤖 Agent Structure
Each solution in the population is represented as an Agent object containing the solution vector and target information. 
class Agent:
    """Individual solution in the population"""
    ID = 0  # Class variable for unique IDs

    def __init__(self, solution=None, target=None, **kwargs):
        self.solution = solution      # np.ndarray: Decision variables
        self.target = target         # Target object: Fitness information
        self.id = self.increase()    # Unique identifier
        self.kwargs = kwargs         # Algorithm-specific attributes
        self.set_kwargs(kwargs)

    @classmethod
    def increase(cls):
        """Generate unique ID for each agent"""
        cls.ID += 1
        return cls.ID

    def is_better_than(self, other_agent, minmax="min"):
        """Compare fitness with another agent"""
        if minmax == "min":
            return self.target.fitness < other_agent.target.fitness
        else:
            return self.target.fitness > other_agent.target.fitness

    def copy(self):
        """Create deep copy of agent"""
        return Agent(self.solution, self.target.copy(), **self.kwargs)

    def update_agent(self, solution, target):
        """Update agent's solution and target"""
        self.solution = solution
        self.target = target

🎯 Target Structure
The Target object encapsulates objective values, weights, and computed fitness for each solution. 
class Target:
    """Encapsulates objective values and fitness calculation"""
    SUPPORTED_ARRAY = [tuple, list, np.ndarray]

    def __init__(self, objectives=None, weights=None):
        self._objectives = None          # List/array of objective values
        self._weights = None            # Weights for multi-objective problems
        self._fitness = None           # Computed fitness value

        self.set_objectives(objectives)
        self.set_weights(weights)
        self.calculate_fitness(self.weights)

    def set_objectives(self, objectives):
        """Set and validate objective values"""
        if isinstance(objectives, (int, float)):
            self._objectives = [objectives]
        elif type(objectives) in self.SUPPORTED_ARRAY:
            self._objectives = list(objectives)
        else:
            raise ValueError("Objectives must be numeric or array-like")

    def calculate_fitness(self, weights):
        """Calculate fitness from objectives and weights"""
        if not (type(weights) in self.SUPPORTED_ARRAY and
                len(weights) == len(self.objectives)):
            weights = len(self.objectives) * (1.,)  # Equal weights

        # For single-objective: fitness = objectives[0]
        # For multi-objective: fitness = weighted sum
        self._fitness = np.dot(weights, self.objectives)

    @property
    def fitness(self):
        """Get computed fitness value"""
        return self._fitness

    @property
    def objectives(self):
        """Get objective values"""
        return self._objectives

    def copy(self):
        """Create copy of target"""
        return Target(self.objectives, self.weights)

def get_sorted_population(pop, minmax="min"):
    """Sort population by fitness"""
    if minmax == "min":
        return sorted(pop, key=lambda agent: agent.target.fitness)
    else:
        return sorted(pop, key=lambda agent: agent.target.fitness, reverse=True)

def get_best_agent(pop, minmax="min"):
    """Get best agent from population"""
    sorted_pop = get_sorted_population(pop, minmax)
    return sorted_pop[0].copy()

def get_worst_agent(pop, minmax="min"):
    """Get worst agent from population"""
    sorted_pop = get_sorted_population(pop, minmax)

class History:
    """Comprehensive tracking of optimization progress"""

    def __init__(self, **kwargs):
        # Solution tracking
        self.list_global_best = []      # Best solution found so far each epoch
        self.list_current_best = []     # Best solution in current epoch
        self.list_global_worst = []     # Worst solution tracking
        self.list_current_worst = []    # Current worst solutions

        # Fitness tracking
        self.list_global_best_fit = []  # Global best fitness values
        self.list_current_best_fit = [] # Current best fitness values

        # Performance metrics
        self.list_diversity = []        # Population diversity each epoch
        self.list_exploration = []      # Exploration percentage
        self.list_exploitation = []     # Exploitation percentage
        self.list_epoch_time = []       # Runtime per epoch

        # Optional: Full population history (memory intensive)
        self.list_population = []       # Complete population each epoch

        # Metadata
        self.epoch = None              # Total epochs completed
        self.log_to = kwargs.get('log_to', 'console')
        self.log_file = kwargs.get('log_file', 'history.txt')

    def store_initial_best_worst(self, best_agent, worst_agent):
        """Store initial population's best and worst agents"""
        self.list_global_best = [best_agent.copy()]
        self.list_current_best = [best_agent.copy()]
        self.list_global_worst = [worst_agent.copy()]
        self.list_current_worst = [worst_agent.copy()]

    def get_global_repeated_times(self, epsilon):
        """Count consecutive epochs with similar global best fitness"""
        count = 0
        for idx in range(len(self.list_global_best) - 1):
            diff = abs(self.list_global_best[idx].target.fitness -
                      self.list_global_best[idx + 1].target.fitness)
            if diff <= epsilon:
                count += 1
            else:
                count = 0
        return count

⚡ Parallel Processing Flow
When parallel modes are enabled, fitness evaluations are distributed across multiple workers. 
def update_target_for_population(self, pop):
    """Update fitness for entire population based on execution mode"""
    pos_list = [agent.solution for agent in pop]

    if self.mode == "process":
        # Multiprocessing with separate processes
        from concurrent.futures import ProcessPoolExecutor
        with ProcessPoolExecutor(max_workers=self.n_workers) as executor:
            list_results = list(executor.map(
                self.get_target_wrapper,
                pos_list
            ))

    elif self.mode == "thread":
        # Multithreading with shared memory
        from concurrent.futures import ThreadPoolExecutor
        with ThreadPoolExecutor(max_workers=self.n_workers) as executor:
            list_results = list(executor.map(
                self.get_target_wrapper,
                pos_list
            ))

    elif self.mode == "swarm":
        # Sequential but independent evaluations (no agent interaction)
        list_results = [self.get_target(pos) for pos in pos_list]

    # Update population with new targets
    for idx, target in enumerate(list_results):
        pop[idx].target = target

    # Increment function evaluation counter
    self.nfe_counter += len(pop)
    return pop

def get_target_wrapper(self, solution):
    """Wrapper for parallel target evaluation"""
    return self.get_target(solution, counted=False)

def track_optimize_process(self):
    """Finalize optimization tracking and calculate metrics"""
    # Set total epochs completed
    self.history.epoch = len(self.history.list_diversity)

    # Calculate exploration/exploitation percentages
    if self.history.list_diversity:
        div_max = np.max(self.history.list_diversity)
        self.history.list_exploration = 100 * (
            np.array(self.history.list_diversity) / div_max
        )
        self.history.list_exploitation = 100 - self.history.list_exploration

    # Clean up history lists (remove initialization entries)
    self.history.list_global_best = self.history.list_global_best[1:]
    self.history.list_current_best = self.history.list_current_best[1:]
    self.history.list_global_worst = self.history.list_global_worst[1:]

# Convergence curve processing
def my_sampling(data, num_points):
    """Sample convergence curve to fixed number of points for visualization"""
    if len(data) <= num_points:
        return data

    # Create evenly spaced indices
    indices = np.linspace(0, len(data) - 1, num_points, dtype=int)
    return [data[i] for i in indices]

# Plotting convergence curves for all algorithms
def plot_convergence_curves(mean_curves, algorithm_names, max_fes,
                          title, save_path):
    """Generate convergence plots comparing multiple algorithms"""
    plt.figure(figsize=(12, 8))

    for idx, (curve, name) in enumerate(zip(mean_curves, algorithm_names)):
        plt.plot(curve, label=name, linewidth=2)

    plt.xlabel('Function Evaluations')
    plt.ylabel('Best Fitness Value (log scale)')
    plt.yscale('log')
    plt.title(title)
    plt.legend()
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.savefig(save_path, dpi=300, bbox_inches='tight')
    plt.close()
